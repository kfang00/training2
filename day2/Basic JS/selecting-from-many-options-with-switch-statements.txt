function caseInSwitch(val) {
  var answer = "";
  // Only change code below this line
  switch(val) {
    case 1:
      var answer = "alpha";
      break;
    case 2:
      var answer = "beta";
      break;
    case 3:
      var answer = "gamma";
      break;
    case 4:
      var answer = "delta";
      break;
  }



  // Only change code above this line
  return answer;
}

caseInSwitch(1);
